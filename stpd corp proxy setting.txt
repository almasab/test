
use proxy settings to access github.com
[http "https://github.com"]
	proxy = http://urlencode(domain\user:psswd)@corp.proxy:port/
	sslVerify = false
	
command
git config --global http.https://github.com.proxy http://urlencode(domain\user:psswd)@corp.proxy:port/
